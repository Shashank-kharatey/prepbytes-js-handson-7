Q.5 You are given a string S consisting of two letters A and D,where each character represent the winner of N games played between Aditya and Danish, where letter A represents the win of Aditya and letter D represents the win of Danish. You need to find out the that which player wins the maximum number of games or there is a draw between them. Note: You have to complete Game_Winner function. No need to take any input.

Ans.5 var Game_Winner = (S) => 
        {
          var countA=0;
          var countD=0;
          for (i=0; i < S.length; i++)
          {
            if (S.charAt(i)==='A')
            {
              countA++;
            }
            else
            {
              countD++;
            }
          }
          if (countA>countD)
          {
            return "Aditya";
          }
          else if (countD>countA)
          {
            return "Danish";
          }
          else 
          {
            return "Draw";
          }
        }
